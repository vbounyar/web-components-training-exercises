Example of a slider



var sliderHost = document.querySelector('.slider'); var shadowRoot = sliderHost.createShadowRoot(); var images = sliderHost.querySelectorAll('img'); var counter = 0; shadowRoot.innerHTML += ''; window.setInterval(function() { images[counter].style.display = 'none'; if(counter < images.length -1) { counter++; } else { counter = 0; } images[counter].style.display = 'block'; }, 1000);
